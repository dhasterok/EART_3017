\begin{framed}
\noindent Derivatives of a few common functions (note $u = u(x)$, $v = v(x)$,
$a$ and $n$ are constants):
\begin{flalign*}
    &\od{}{x}(a) = 0 &
    &\od{}{x}(x) = 1 \\
    &\od{}{x}(au) = a \od{u}{x} &
    &\od{}{x}(u + v) = \od{u}{x} + \od{v}{x} \\
    &\od{}{x}(uv) = v\od{u}{x} + u\od{v}{x} &
    &\od{}{x}\left(\frac{u}{v}\right) = v^{-2}\left(v\od{u}{x} - u\od{v}{x}\right) \\
    &\od{}{x}u(v) = \od{u}{v}\od{v}{x} &
    &\od{y}{x} = \left( \od{x}{y} \right)^{-1} \\
    &\od{}{x}(u^n) = n u^{n-1} \od{u}{x} &
    &\od{}{x}\left(\frac{1}{u^n}\right) = -\frac{n}{u^{n + 1}} \od{u}{x} \\
    &\od{}{x}(e^u) = e^u \od{u}{x} & 
    &\od{}{x}(\log u) = \frac{1}{u} \od{u}{x},\; u > 0 \\
    &\od{}{x}(\sin u) = \cos u \od{u}{x} &
    &\od{}{x}(\cos u) = -\sin u \od{u}{x}
\end{flalign*}
\end{framed}
