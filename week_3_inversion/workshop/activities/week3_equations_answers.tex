\subsection*{Answer key (concise)}
\begin{enumerate}[label=A\arabic*.,itemsep=6pt]
  \item \emph{Bilinear} in $(\Delta\rho,H)$; \emph{linear} if one is known; or reparameterize $m=\Delta\rho H$.
  \item \emph{Non-linear} in $(R,z)$; \emph{linear} in $\Delta\rho$ (or $M$). Linearize via Jacobian about $p_0$.
  \item \emph{Linear} in $[a,b,c]$ for fixed $n$ and known $X$; \emph{non-linear} in $n$ (block LS + GN).
  \item \emph{Linear} in $k_0$ if $(A,B,c)$ known; \emph{non-linear} jointly. Often fix $(A,B)$.
  \item \emph{Linear} relation between $h$ and $\Delta t$ for known densities; becomes \emph{bilinear} if densities are unknown.
  \item \emph{Linear} (didactic column model) in the stated terms.
  \item \emph{Linear} in amplitude $C$ (geometry fixed); \emph{non-linear} in $(x_1,x_2,z)$.
  \item \emph{Linear} in heat production set $\{H_k\}$ for fixed $\{k_k\}$ once interface/BC constraints are applied; \emph{non-linear} if $k$ depends on $T$.
  \item \emph{Linear} in porosity $\phi$ for known $(\rho_m,\rho_f)$.
  \item Non-linear. Take logs: $\ln\rho = \ln a + b\,\ln v$ (linear in $[\ln a, b]$).
  \item \emph{Linear} in $\Delta\rho_i$ (geometry fixed).
\end{enumerate}
